about:config

  • accessibility.typeaheadfind.enablesound = false—stfu
  • accessibility.typeaheadfind.flashBar = 0
  • accessibility.typeaheadfind.soundURL = (blank)—stfu
  • browser.blink_allowed = false—disable blink element
  • browser.open_external = 3
  • browser.open_newwindow = 1
  • browser.open_newwindow.ui = 1
  • browser.tabs.showSingleWindowModePrefs = true—show the magical hidden preferences!
  • browser.xul.error_pages.enabled = true—an error page instead of an error dialog—false until 1.1
  • image.animation_mode = none—show one frame of animated images
  • intl.accept_charsets = utf-8,iso-8859-1;qs=0.9,*;qs=0.5—UTF-8, please
  • network.IDN_show_punycode = false—allow real IDNs in all their UTF glory
  • network.http.accept.default = application/xhtml+xml,text/html;q=0.09,application/xml;q=0.08,text/xml;q=0.07,image/png;q=0.06,image/jpeg;q=0.05,text/plain;q=0.03,*/*;q=0.01—much better than the stupid default
  • network.http.pipelining = true—go faster!
  • network.http.pipelining.maxrequests = 8—How much faster‽ 8 faster!
  • network.standard-url.escape-utf8 = false—show the UTF-8 in the path
  • ui.key.generalAccessKey = 0—disable accesskeys (because they suck)
